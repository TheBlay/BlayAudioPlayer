CREATE TABLE `musicplayer`.`musicas`
 (`id` INT NOT NULL AUTO_INCREMENT ,
 `titulo` VARCHAR(150) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL ,
 `artista` VARCHAR(150) NULL ,
 `album` VARCHAR(150) NULL ,
 `caminho_arquivo` VARCHAR(255) NOT NULL ,
 `caminho_capa` VARCHAR(255) DEFAULT 'capa_default' ,
 PRIMARY KEY (`id`)
) ENGINE = InnoDB CHARSET=utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE `musicplayer`.`tags` (
     `id` INT NOT NULL AUTO_INCREMENT,
     `nome` VARCHAR(50) NOT NULL,
     PRIMARY KEY (`id`),          -- Correção aqui: usando crases ou apenas (id)
     UNIQUE INDEX `idx_nome_unico` (`nome`) -- Uma forma mais elegante de declarar o UNIQUE
) ENGINE = InnoDB CHARSET=utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE `musica_tags` (
  `musica_id` INT NOT NULL,
  `tag_id` INT NOT NULL,
  
  -- 1. Criação da Chave Primária Composta (Garante unicidade do vínculo)
  PRIMARY KEY (`musica_id`, `tag_id`),
  
  -- 2. Índice individual para otimizar a busca reversa (filtrar músicas por tag)
  INDEX `idx_tag_id` (`tag_id`),
  
  -- 3. Chave Estrangeira ligando à tabela 'musicas'
  CONSTRAINT `fk_musica` 
    FOREIGN KEY (`musica_id`) REFERENCES `musicas` (`id`) 
    ON DELETE CASCADE ON UPDATE CASCADE,
    
  -- 4. Chave Estrangeira ligando à tabela 'tags'
  CONSTRAINT `fk_tag` 
    FOREIGN KEY (`tag_id`) REFERENCES `tags` (`id`) 
    ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
